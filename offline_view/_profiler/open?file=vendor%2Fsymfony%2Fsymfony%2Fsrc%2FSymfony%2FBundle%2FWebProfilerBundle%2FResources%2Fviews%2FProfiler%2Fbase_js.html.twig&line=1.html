  448.                             }
  450.                             /* get the full list of tabs through the parent of the active tab element */
  451.                             var tabNavigation = activeTab.parentNode.children;
  452.                             for (var k = 0; k < tabNavigation.length; k++) {
  453.                                 var tabId = tabNavigation[k].getAttribute('data-tab-id');
  454.                                 document.getElementById(tabId).className = 'hidden';
  455.                                 removeClass(tabNavigation[k], 'active');
  456.                             }
  458.                             addClass(activeTab, 'active');
  459.                             var activeTabId = activeTab.getAttribute('data-tab-id');
  460.                             document.getElementById(activeTabId).className = 'block';
  461.                         });
  462.                     }
  464.                     tabGroups[i].setAttribute('data-processed', 'true');
  465.                 }
  466.             },
  468.             createToggles: function() {
  469.                 var toggles = document.querySelectorAll('.sf-toggle:not([data-processed=true])');
  471.                 for (var i = 0; i < toggles.length; i++) {
  472.                     var elementSelector = toggles[i].getAttribute('data-toggle-selector');
  473.                     var element = document.querySelector(elementSelector);
  475.                     addClass(element, 'sf-toggle-content');
  477.                     if (toggles[i].hasAttribute('data-toggle-initial') && toggles[i].getAttribute('data-toggle-initial') == 'display') {
  478.                         addClass(toggles[i], 'sf-toggle-on');
  479.                         addClass(element, 'sf-toggle-visible');
  480.                     } else {
  481.                         addClass(toggles[i], 'sf-toggle-off');
  482.                         addClass(element, 'sf-toggle-hidden');
  483.                     }
  485.                     addEventListener(toggles[i], 'click', function(e) {
  486.                         e.preventDefault();
  488.                         if ('' !== window.getSelection().toString()) {
  489.                             /* Don't do anything on text selection */
  490.                             return;
  491.                         }
  493.                         var toggle = e.target || e.srcElement;
  495.                         /* needed because when the toggle contains HTML contents, user can click */
  496.                         /* on any of those elements instead of their parent '.sf-toggle' element */
  497.                         while (!hasClass(toggle, 'sf-toggle')) {
  498.                             toggle = toggle.parentNode;
  499.                         }
  501.                         var element = document.querySelector(toggle.getAttribute('data-toggle-selector'));
  503.                         toggleClass(toggle, 'sf-toggle-on');
  504.                         toggleClass(toggle, 'sf-toggle-off');
  505.                         toggleClass(element, 'sf-toggle-hidden');
  506.                         toggleClass(element, 'sf-toggle-visible');
  508.                         /* the toggle doesn't change its contents when clicking on it */
  509.                         if (!toggle.hasAttribute('data-toggle-alt-content')) {
  510.                             return;
  511.                         }
  513.                         if (!toggle.hasAttribute('data-toggle-original-content')) {
  514.                             toggle.setAttribute('data-toggle-original-content', toggle.innerHTML);
  515.                         }
  517.                         var currentContent = toggle.innerHTML;
  518.                         var originalContent = toggle.getAttribute('data-toggle-original-content');
  519.                         var altContent = toggle.getAttribute('data-toggle-alt-content');
  520.                         toggle.innerHTML = currentContent !== altContent ? altContent : originalContent;
  521.                     });
  523.                     /* Prevents from disallowing clicks on links inside toggles */
  524.                     var toggleLinks = toggles[i].querySelectorAll('a');
  525.                     for (var j = 0; j < toggleLinks.length; j++) {
  526.                         addEventListener(toggleLinks[j], 'click', function(e) {
  527.                             e.stopPropagation();
  528.                         });
  529.                     }
  531.                     toggles[i].setAttribute('data-processed', 'true');
  532.                 }
  533.             }
  534.         };
  535.     })();
  537.     Sfjs.addEventListener(document, 'DOMContentLoaded', function() {
  538.         Sfjs.createTabs();
  539.         Sfjs.createToggles();
