  506.                     throw new \LogicException(sprintf('Bundle "%s" can not extend itself.', $name));
  507.                 }
  508.                 $directChildren[$parentName] = $name;
  509.             } else {
  510.                 $topMostBundles[$name] = $bundle;
  511.             }
  512.         }
  514.         // look for orphans
  515.         if (!empty($directChildren) && \count($diff = array_diff_key($directChildren, $this->bundles))) {
  516.             $diff = array_keys($diff);
  518.             throw new \LogicException(sprintf('Bundle "%s" extends bundle "%s", which is not registered.', $directChildren[$diff[0]], $diff[0]));
  519.         }
  521.         // inheritance
  522.         $this->bundleMap = array();
  523.         foreach ($topMostBundles as $name => $bundle) {
  524.             $bundleMap = array($bundle);
  525.             $hierarchy = array($name);
  527.             while (isset($directChildren[$name])) {
  528.                 $name = $directChildren[$name];
  529.                 array_unshift($bundleMap, $this->bundles[$name]);
  530.                 $hierarchy[] = $name;
  531.             }
  533.             foreach ($hierarchy as $hierarchyBundle) {
  534.                 $this->bundleMap[$hierarchyBundle] = $bundleMap;
  535.                 array_pop($bundleMap);
  536.             }
  537.         }
  538.     }

  541.      * The extension point similar to the Bundle::build() method.
  542.      *
  543.      * Use this method to register compiler passes and manipulate the container during the building process.
  544.      */
  545.     protected function build(ContainerBuilder $container)
  546.     {
  547.     }

  552.      * @return string The container class

  554.     protected function getContainerClass()

  557.     }
  559.     /**
  560.      * Gets the container's base class.
  561.      *

  563.      *
  564.      * @return string

  569.     }
  571.     /**
  572.      * Initializes the service container.
  573.      *
  574.      * The cached version of the service container is used when fresh, otherwise the
  575.      * container is built.
  576.      */
  577.     protected function initializeContainer()
  578.     {
  579.         $class = $this->getContainerClass();
  580.         $cacheDir = $this->warmupDir ?: $this->getCacheDir();
  581.         $cache = new ConfigCache($cacheDir.'/'.$class.'.php', $this->debug);
  582.         $oldContainer = null;
  583.         if ($fresh = $cache->isFresh()) {
  584.             // Silence E_WARNING to ignore "include" failures - don't use "@" to prevent silencing fatal errors
  585.             $errorLevel = error_reporting(\E_ALL ^ \E_WARNING);
  586.             $fresh = $oldContainer = false;
  587.             try {
  588.                 if (file_exists($cache->getPath()) && \is_object($this->container = include $cache->getPath())) {
  589.                     $this->container->set('kernel', $this);
  590.                     $oldContainer = $this->container;
  591.                     $fresh = true;
  592.                 }
  593.             } catch (\Throwable $e) {
  594.             } catch (\Exception $e) {
  595.             } finally {
  596.                 error_reporting($errorLevel);
  597.             }
  598.         }
  600.         if ($fresh) {
  601.             return;
  602.         }
  604.         if ($this->debug) {
  605.             $collectedLogs = array();
  606.             $previousHandler = \defined('PHPUNIT_COMPOSER_INSTALL');
  607.             $previousHandler = $previousHandler ?: set_error_handler(function ($type, $message, $file, $line) use (&$collectedLogs, &$previousHandler) {
  608.                 if (E_USER_DEPRECATED !== $type && E_DEPRECATED !== $type) {
  609.                     return $previousHandler ? $previousHandler($type, $message, $file, $line) : false;
  610.                 }
  612.                 if (isset($collectedLogs[$message])) {
  613.                     ++$collectedLogs[$message]['count'];
  615.                     return;
  616.                 }
  618.                 $backtrace = debug_backtrace(DEBUG_BACKTRACE_IGNORE_ARGS, 3);
  619.                 // Clean the trace by removing first frames added by the error handler itself.
  620.                 for ($i = 0; isset($backtrace[$i]); ++$i) {
  621.                     if (isset($backtrace[$i]['file'], $backtrace[$i]['line']) && $backtrace[$i]['line'] === $line && $backtrace[$i]['file'] === $file) {
  622.                         $backtrace = \array_slice($backtrace, 1 + $i);
  623.                         break;
  624.                     }
  625.                 }
  627.                 $collectedLogs[$message] = array(
  628.                     'type' => $type,
  629.                     'message' => $message,
  630.                     'file' => $file,
  631.                     'line' => $line,
  632.                     'trace' => $backtrace,
  633.                     'count' => 1,
  634.                 );
  635.             });
  636.         }
  638.         try {
  639.             $container = null;
  640.             $container = $this->buildContainer();
  641.             $container->compile();
  642.         } finally {
  643.             if ($this->debug && true !== $previousHandler) {
  644.                 restore_error_handler();
  646.                 file_put_contents($cacheDir.'/'.$class.'Deprecations.log', serialize(array_values($collectedLogs)));
  647.                 file_put_contents($cacheDir.'/'.$class.'Compiler.log', null !== $container ? implode("\n", $container->getCompiler()->getLog()) : '');
  648.             }
  649.         }
  651.         if (null === $oldContainer && file_exists($cache->getPath())) {
  652.             $errorLevel = error_reporting(\E_ALL ^ \E_WARNING);
  653.             try {
  654.                 $oldContainer = include $cache->getPath();
  655.             } catch (\Throwable $e) {
  656.             } catch (\Exception $e) {
  657.             } finally {
  658.                 error_reporting($errorLevel);
  659.             }
  660.         }
  661.         $oldContainer = \is_object($oldContainer) ? new \ReflectionClass($oldContainer) : false;
  663.         $this->dumpContainer($cache, $container, $class, $this->getContainerBaseClass());
  664.         $this->container = require $cache->getPath();
  665.         $this->container->set('kernel', $this);
  667.         if ($oldContainer && \get_class($this->container) !== $oldContainer->name) {
  668.             // Because concurrent requests might still be using them,
  669.             // old container files are not removed immediately,
  670.             // but on a next dump of the container.
  671.             static $legacyContainers = array();
  672.             $oldContainerDir = \dirname($oldContainer->getFileName());
  673.             $legacyContainers[$oldContainerDir.'.legacy'] = true;
  674.             foreach (glob(\dirname($oldContainerDir).\DIRECTORY_SEPARATOR.'*.legacy') as $legacyContainer) {
  675.                 if (!isset($legacyContainers[$legacyContainer]) && @unlink($legacyContainer)) {
  676.                     (new Filesystem())->remove(substr($legacyContainer, 0, -7));
  677.                 }
  678.             }
  680.             touch($oldContainerDir.'.legacy');
  681.         }
  683.         if ($this->container->has('cache_warmer')) {
  684.             $this->container->get('cache_warmer')->warmUp($this->container->getParameter('kernel.cache_dir'));
  685.         }
  686.     }
  688.     /**
  689.      * Returns the kernel parameters.
  690.      *
  691.      * @return array An array of kernel parameters
  692.      */
  693.     protected function getKernelParameters()
  694.     {
  695.         $bundles = array();
  696.         $bundlesMetadata = array();
  698.         foreach ($this->bundles as $name => $bundle) {
  699.             $bundles[$name] = \get_class($bundle);
  700.             $bundlesMetadata[$name] = array(
  701.                 'parent' => $bundle->getParent(),
  702.                 'path' => $bundle->getPath(),
  703.                 'namespace' => $bundle->getNamespace(),
  704.             );
  705.         }
  707.         return array_merge(
  708.             array(
  709.                 'kernel.root_dir' => realpath($this->rootDir) ?: $this->rootDir,
  710.                 'kernel.project_dir' => realpath($this->getProjectDir()) ?: $this->getProjectDir(),
  711.                 'kernel.environment' => $this->environment,
  712.                 'kernel.debug' => $this->debug,
  713.                 'kernel.name' => $this->name,
  714.                 'kernel.cache_dir' => realpath($cacheDir = $this->warmupDir ?: $this->getCacheDir()) ?: $cacheDir,
  715.                 'kernel.logs_dir' => realpath($this->getLogDir()) ?: $this->getLogDir(),
  716.                 'kernel.bundles' => $bundles,
  717.                 'kernel.bundles_metadata' => $bundlesMetadata,
  718.                 'kernel.charset' => $this->getCharset(),
  719.                 'kernel.container_class' => $this->getContainerClass(),
  720.             ),
  721.             $this->getEnvParameters(false)
  722.         );
  723.     }
  725.     /**
  726.      * Gets the environment parameters.
  727.      *
  728.      * Only the parameters starting with "SYMFONY__" are considered.
  729.      *
  730.      * @return array An array of parameters
  731.      *
  732.      * @deprecated since version 3.3, to be removed in 4.0
  733.      */
  734.     protected function getEnvParameters()
  735.     {
  736.         if (0 === \func_num_args() || func_get_arg(0)) {
  737.             @trigger_error(sprintf('The "%s()" method is deprecated as of 3.3 and will be removed in 4.0. Use the %%env()%% syntax to get the value of any environment variable from configuration files instead.', __METHOD__), E_USER_DEPRECATED);
  738.         }
  740.         $parameters = array();
  741.         foreach ($_SERVER as $key => $value) {
  742.             if (0 === strpos($key, 'SYMFONY__')) {
  743.                 @trigger_error(sprintf('The support of special environment variables that start with SYMFONY__ (such as "%s") is deprecated as of 3.3 and will be removed in 4.0. Use the %%env()%% syntax instead to get the value of environment variables in configuration files.', $key), E_USER_DEPRECATED);
  744.                 $parameters[strtolower(str_replace('__', '.', substr($key, 9)))] = $value;
  745.             }
  746.         }
  748.         return $parameters;
  749.     }
  751.     /**
  752.      * Builds the service container.
  753.      *
  754.      * @return ContainerBuilder The compiled service container
  755.      *
  756.      * @throws \RuntimeException
  757.      */
  758.     protected function buildContainer()
  759.     {
  760.         foreach (array('cache' => $this->warmupDir ?: $this->getCacheDir(), 'logs' => $this->getLogDir()) as $name => $dir) {
  761.             if (!is_dir($dir)) {
  762.                 if (false === @mkdir($dir, 0777, true) && !is_dir($dir)) {
  763.                     throw new \RuntimeException(sprintf("Unable to create the %s directory (%s)\n", $name, $dir));
  764.                 }
  765.             } elseif (!is_writable($dir)) {
  766.                 throw new \RuntimeException(sprintf("Unable to write in the %s directory (%s)\n", $name, $dir));
  767.             }
  768.         }
  770.         $container = $this->getContainerBuilder();
  771.         $container->addObjectResource($this);
  772.         $this->prepareContainer($container);
  774.         if (null !== $cont = $this->registerContainerConfiguration($this->getContainerLoader($container))) {
  775.             $container->merge($cont);
  776.         }
  778.         $container->addCompilerPass(new AddAnnotatedClassesToCachePass($this));
  779.         $container->addResource(new EnvParametersResource('SYMFONY__'));
  781.         return $container;
  782.     }
  784.     /**
  785.      * Prepares the ContainerBuilder before it is compiled.
  786.      */
  787.     protected function prepareContainer(ContainerBuilder $container)
  788.     {
  789.         $extensions = array();
  790.         foreach ($this->bundles as $bundle) {
  791.             if ($extension = $bundle->getContainerExtension()) {
  792.                 $container->registerExtension($extension);
  793.             }
  795.             if ($this->debug) {
  796.                 $container->addObjectResource($bundle);
  797.             }
  798.         }
  800.         foreach ($this->bundles as $bundle) {
  801.             $bundle->build($container);
  802.         }
  804.         $this->build($container);
  806.         foreach ($container->getExtensions() as $extension) {
  807.             $extensions[] = $extension->getAlias();
  808.         }
  810.         // ensure these extensions are implicitly loaded
  811.         $container->getCompilerPassConfig()->setMergePass(new MergeExtensionConfigurationPass($extensions));
  812.     }
  814.     /**
  815.      * Gets a new ContainerBuilder instance used to build the service container.
  816.      *
  817.      * @return ContainerBuilder
  818.      */
  819.     protected function getContainerBuilder()
  820.     {
  821.         $container = new ContainerBuilder();
  822.         $container->getParameterBag()->add($this->getKernelParameters());
  824.         if ($this instanceof CompilerPassInterface) {
  825.             $container->addCompilerPass($this, PassConfig::TYPE_BEFORE_OPTIMIZATION, -10000);
  826.         }
  827.         if (class_exists('ProxyManager\Configuration') && class_exists('Symfony\Bridge\ProxyManager\LazyProxy\Instantiator\RuntimeInstantiator')) {
  828.             $container->setProxyInstantiator(new RuntimeInstantiator());
  829.         }
  831.         return $container;
  832.     }
  834.     /**
  835.      * Dumps the service container to PHP code in the cache.
  836.      *
  837.      * @param ConfigCache      $cache     The config cache
  838.      * @param ContainerBuilder $container The service container
  839.      * @param string           $class     The name of the class to generate
  840.      * @param string           $baseClass The name of the container's base class
  841.      */
  842.     protected function dumpContainer(ConfigCache $cache, ContainerBuilder $container, $class, $baseClass)
  843.     {
  844.         // cache the container
  845.         $dumper = new PhpDumper($container);
  847.         if (class_exists('ProxyManager\Configuration') && class_exists('Symfony\Bridge\ProxyManager\LazyProxy\PhpDumper\ProxyDumper')) {
  848.             $dumper->setProxyDumper(new ProxyDumper());
  849.         }
  851.         $content = $dumper->dump(array(
  852.             'class' => $class,
  853.             'base_class' => $baseClass,
  854.             'file' => $cache->getPath(),
  855.             'as_files' => true,
  856.             'debug' => $this->debug,
  857.             'inline_class_loader_parameter' => \PHP_VERSION_ID >= 70000 && !$this->loadClassCache && !class_exists(ClassCollectionLoader::class, false) ? 'container.dumper.inline_class_loader' : null,
  858.             'build_time' => $container->hasParameter('kernel.container_build_time') ? $container->getParameter('kernel.container_build_time') : time(),
  859.         ));
  861.         $rootCode = array_pop($content);
  862.         $dir = \dirname($cache->getPath()).'/';
  863.         $fs = new Filesystem();
  865.         foreach ($content as $file => $code) {
  866.             $fs->dumpFile($dir.$file, $code);
  867.             @chmod($dir.$file, 0666 & ~umask());
  868.         }
  869.         @unlink(\dirname($dir.$file).'.legacy');
  871.         $cache->write($rootCode, $container->getResources());
  872.     }
  874.     /**
  875.      * Returns a loader for the container.
  876.      *
  877.      * @return DelegatingLoader The loader
  878.      */
  879.     protected function getContainerLoader(ContainerInterface $container)
  880.     {
  881.         $locator = new FileLocator($this);
  882.         $resolver = new LoaderResolver(array(
  883.             new XmlFileLoader($container, $locator),
  884.             new YamlFileLoader($container, $locator),
  885.             new IniFileLoader($container, $locator),
  886.             new PhpFileLoader($container, $locator),
  887.             new GlobFileLoader($container, $locator),
  888.             new DirectoryLoader($container, $locator),
  889.             new ClosureLoader($container),
  890.         ));
  892.         return new DelegatingLoader($resolver);
  893.     }
  895.     /**
  896.      * Removes comments from a PHP source string.
  897.      *
  898.      * We don't use the PHP php_strip_whitespace() function
  899.      * as we want the content to be readable and well-formatted.
  900.      *
  901.      * @param string $source A PHP string
  902.      *
  903.      * @return string The PHP string with the comments removed
  904.      */
  905.     public static function stripComments($source)
  906.     {
  907.         if (!\function_exists('token_get_all')) {
  908.             return $source;
  909.         }
  911.         $rawChunk = '';
  912.         $output = '';
  913.         $tokens = token_get_all($source);
  914.         $ignoreSpace = false;
  915.         for ($i = 0; isset($tokens[$i]); ++$i) {
  916.             $token = $tokens[$i];
  917.             if (!isset($token[1]) || 'b"' === $token) {
  918.                 $rawChunk .= $token;
  919.             } elseif (T_START_HEREDOC === $token[0]) {
  920.                 $output .= $rawChunk.$token[1];
  921.                 do {
  922.                     $token = $tokens[++$i];
  923.                     $output .= isset($token[1]) && 'b"' !== $token ? $token[1] : $token;
  924.                 } while (T_END_HEREDOC !== $token[0]);
  925.                 $rawChunk = '';
  926.             } elseif (T_WHITESPACE === $token[0]) {
  927.                 if ($ignoreSpace) {
  928.                     $ignoreSpace = false;
  930.                     continue;
  931.                 }
  933.                 // replace multiple new lines with a single newline
  934.                 $rawChunk .= preg_replace(array('/\n{2,}/S'), "\n", $token[1]);
  935.             } elseif (\in_array($token[0], array(T_COMMENT, T_DOC_COMMENT))) {
  936.                 $ignoreSpace = true;
  937.             } else {
  938.                 $rawChunk .= $token[1];
  940.                 // The PHP-open tag already has a new-line
  941.                 if (T_OPEN_TAG === $token[0]) {
  942.                     $ignoreSpace = true;
  943.                 }
  944.             }
